모든 방명록

{# 아래는 Jinja 반복문: messages 리스트를 순회 #} {% for m in messages %} {# m[1]: 이름, m[2]: 메시지 #} {# 필터링 없이 출력 → Stored XSS 취약점 존재 (과제/테스트 목적) #}

{{ m[1] }}: {{ m[2]|safe }}

{% endfor %} {# 참고: HTML 주석 안에 {{ ... }} 또는 {% ... %} 같은 Jinja 표기를 넣으면 Jinja가 그 안도 파싱하려 하기 때문에 오류가 발생할 수 있음 템플릿 설명은 Jinja 주석 {# ... #}을 사용 #}
